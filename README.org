* tinydis

A tiny and anonymous discussion plugin implented by Rust.

** dev

#+begin_src bash
  cargo install cargo-leptos --locked
  cargo leptos watch
  cargo leptos build
#+end_src

** usage 


#+begin_src bash
  export TINYDIS_SMTP_USERNAME=your_email_to_send_review_message
  export TINYDIS_SMTP_PASSWORD=your_password_to_send_review_message   # an App password is prefered to regular parssword
  export TINYDIS_SMTP_HOST=your_smtp_host
  export TINYDIS_SMTP_PORT=your_smtp_port
  export TINYDIS_ADMIN_EMAIL=admin_email_to_receive_review_message
  export TINYDIS_ALLOWED_ORIGINS="*" # ALLOWED ORIGIN for backend server, such as "http://127.0.0.1:8080,http://127.0.0.1:9999"
  export LEPTOS_SITE_ADDR="127.0.0.1:3000" # backend site ip:port
  export TINYDIS_SERVER_ADDR="http://${LEPTOS_SITE_ADDR}" # use https or http
#+end_src

#+begin_src bash
  cargo install --locked cargo-leptos
  cargo leptos build --release && mv target/site/assets/tinydis.wasm  target/site/assets/tinydis_bg.wasm
  cp -r ./target/site/assets YOUR_SITE_ROOT_PATH
#+end_src

html head:

#+begin_src html
  <link rel="stylesheet" href="/assets/tinydis.css" />
#+end_src

html body:

#+begin_src html
  <div id="tinydis" data-server-url="TINYDIS_SERVER_ADDR"></div>
  <script type="module">
    import init from '/assets/tinydis.js';
    init();
  </script>
#+end_src

Or you can explictly set data-page-id
#+begin_src html
  <div id="tinydis" data-page-id="YOUR_PAGE_ID" data-server-url="TINYDIS_SERVER_ADDR"></div>
  <script type="module">
    import init from '/assets/tinydis.js';
    init();
  </script>
#+end_src
